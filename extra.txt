        // Function to display featured PDFs (limited number)
        function displayFeaturedPDFs() {
            const featuredPdfList = document.getElementById('featuredPdfList');

            // Show only 5 featured PDFs
            const featuredPDFs = pdfData.slice(0, 5);

            featuredPdfList.innerHTML = '';

            featuredPDFs.forEach(pdf => {
                const pdfItem = document.createElement('div');
                pdfItem.className = 'pdf-item';
                pdfItem.onclick = () => openPdf(pdf.path, pdf.fileName);

                pdfItem.innerHTML = `
                    <div class="pdf-icon">
                        <i class="fas fa-file-pdf"></i>
                    </div>
                    <div class="pdf-info">
                        <h4>${pdf.title}</h4>
                        <p>${branchNames[pdf.branch]} - Semester ${pdf.semester}</p>
                        <div class="pdf-meta">
                            <span class="pdf-tag">${pdf.year}</span>
                            <span class="pdf-tag">${subjectNames[pdf.subject]}</span>
                        </div>
                    </div>
                `;

                featuredPdfList.appendChild(pdfItem);
            });
        }

        // Function to display all PDFs
        function displayAllPDFs(pdfs) {
            const allPdfList = document.getElementById('allPdfList');
            const noResults = document.getElementById('noResults');
            const searchInfo = document.getElementById('searchInfo');

            if (pdfs.length === 0) {
                allPdfList.style.display = 'none';
                noResults.style.display = 'block';
                searchInfo.textContent = '';
                return;
            }

            allPdfList.style.display = 'grid';
            noResults.style.display = 'none';

            // Show search info
            if (pdfs.length === pdfData.length) {
                searchInfo.textContent = 'Showing 5 featured papers. Use filters to search for specific papers.';
            } else {
                searchInfo.textContent = `Found ${pdfs.length} paper(s) matching your search criteria.`;
            }

            allPdfList.innerHTML = '';

            pdfs.forEach(pdf => {
                const pdfItem = document.createElement('div');
                pdfItem.className = 'pdf-item';
                pdfItem.onclick = () => openPdf(pdf.path, pdf.fileName);

                pdfItem.innerHTML = `
                    <div class="pdf-icon">
                        <i class="fas fa-file-pdf"></i>
                    </div>
                    <div class="pdf-info">
                        <h4>${pdf.title}</h4>
                        <p>${branchNames[pdf.branch]} - Semester ${pdf.semester}</p>
                        <div class="pdf-meta">
                            <span class="pdf-tag">${pdf.year}</span>
                            <span class="pdf-tag">${subjectNames[pdf.subject]}</span>
                        </div>
                    </div>
                `;

                allPdfList.appendChild(pdfItem);
            });
        }

        // Function to populate subject dropdown
        function populateSubjectDropdown() {
            const subjectSelect = document.getElementById('subject');
            const uniqueSubjects = [...new Set(pdfData.map(pdf => pdf.subject))];

            uniqueSubjects.forEach(subject => {
                const option = document.createElement('option');
                option.value = subject;
                option.textContent = subjectNames[subject];
                subjectSelect.appendChild(option);
            });
        }

        // Function to apply filters
        function applyFilters() {
            const branch = document.getElementById('branch').value;
            const semester = document.getElementById('semester').value;
            const year = document.getElementById('year').value;
            const subject = document.getElementById('subject').value;

            const filteredPDFs = pdfData.filter(pdf => {
                return (branch === 'all' || pdf.branch === branch) &&
                    (semester === 'all' || pdf.semester === semester) &&
                    (year === 'all' || pdf.year === year) &&
                    (subject === 'all' || pdf.subject === subject);
            });

            displayAllPDFs(filteredPDFs);

            // Scroll to results
            document.getElementById('all-pdfs').scrollIntoView({ behavior: 'smooth' });
        }

        // Function to reset filters
        function resetFilters() {
            document.getElementById('branch').value = 'all';
            document.getElementById('semester').value = 'all';
            document.getElementById('year').value = 'all';
            document.getElementById('subject').value = 'all';

            // Show only 5 papers when reset
            displayAllPDFs(pdfData.slice(0, 5));
        }

        // Function to open PDF
        function openPdf(pdfPath, fileName) {
            if (pdfPath === '#') {
                alert(`PDF "${fileName}" link not configured yet. Please add your PDF path in the code.`);
                return;
            }
            window.open(pdfPath, '_blank');
        }

        // Mobile menu toggle
        document.getElementById('menuToggle').addEventListener('click', function () {
            const nav = document.getElementById('mainNav');
            nav.classList.toggle('active');
        });

        // Smooth scrolling for navigation links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();

                const targetId = this.getAttribute('href');
                if (targetId === '#') return;

                const targetElement = document.querySelector(targetId);
                if (targetElement) {
                    window.scrollTo({
                        top: targetElement.offsetTop - 80,
                        behavior: 'smooth'
                    });
                }

                // Close mobile menu after clicking
                const nav = document.getElementById('mainNav');
                nav.classList.remove('active');
            });
        });

        // Initialize the page
        document.addEventListener('DOMContentLoaded', function () {
            populateSubjectDropdown();
            displayFeaturedPDFs();
            // Initially show only 5 papers
            displayAllPDFs(pdfData.slice(0, 5));
        });

        // Add animation on scroll
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('fade-in');
                }
            });
        }, observerOptions);

        document.querySelectorAll('.pdf-item, .stat-item').forEach(el => {
            observer.observe(el);
        });
